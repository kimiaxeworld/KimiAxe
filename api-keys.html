# Send Bulk SMS
curl -X POST https://api.kimiaxe.com/v1/sms/send \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "to": ["+919876543210", "+919876543211"],
    "message": "Hello from KimiAxe!",
    "sender": "AXESMS"
  }'

# Shorten a URL
curl -X POST https://api.kimiaxe.com/v1/links/shorten \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -d '{"url": "https://your-long-url.com"}'
